Array.prototype.igualdad=function(palabra){
  
  igual=0;
  
  for(var i=0;i<this.length;i++){
    if(this[i]==palabra){
      igual++;
    }
  }
  
  console.log('hay '+igual+' palabras repetidas');
  
}
